    function updateProgress() {

      if (count === 0) {

        tracker.textContent = `Clicks: ${count}/${total}`;

      } else {

        tracker.textContent = `i ${"love ".repeat(count)}you`;

      bar.style.width = pct + '%';

    }


    function placeSparkles() {

      const wrap = document.getElementById('sparkles');

      for (let i = 0; i < 12; i++) {

        const s = document.createElement('span');

        s.style.left = Math.random() * 100 + '%';

        s.style.top = Math.random() * 100 + '%';

        const size = 6 + Math.random() * 8;

        s.style.width = size + 'px';

        s.style.height = size + 'px';

        s.style.opacity = 0.6 + Math.random() * 0.5;

        s.style.animationDuration = (8 + Math.random() * 8) + 's';

        wrap.appendChild(s);

      }

    }

    placeSparkles();


    function resizeCanvas() {

      const rect = card.getBoundingClientRect();

      cvs.width = rect.width;

    function makeConfettiBurst() {

      const pieces = [];

      for (let i = 0; i < 70; i++) {

        pieces.push({

          x: cvs.width / 2,

          y: cvs.height / 2,

          vx: (Math.random() - 0.5) * 6,

          vy: (Math.random() * -6) - 2,

          r: 4 + Math.random() * 6,

          color: ['#d4b14f', '#ffd966', '#f7c6ff', '#c69df0', '#fff2cc'][Math.floor(Math.random() * 5)],

          rot: Math.random() * 360

        });

      }

      const lifetime = 120;

      let t = 0;

      function frame() {

        ctx.clearRect(0, 0, cvs.width, cvs.height);

        for (let p of pieces) {

          p.vy += 0.15;

          p.x += p.vx;

          p.y += p.vy;

          p.rot += 6;

          ctx.save();

          ctx.translate(p.x, p.y);

          ctx.rotate(p.rot * Math.PI / 180);

          ctx.fillStyle = p.color;

          ctx.fillRect(-p.r / 2, -p.r / 2, p.r, p.r);

          ctx.restore();

        }

        t++;

        if (t < lifetime) {

          requestAnimationFrame(frame);

        } else {

          ctx.clearRect(0, 0, cvs.width, cvs.height);

        }

      }

      requestAnimationFrame(frame);

    }

      }

    });
